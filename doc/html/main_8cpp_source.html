66 bool webpage_crawler(const Socket::Url &url, const path &output) {
67  // Download given URL page
68  std::stringstream page_stream{};
69  auto res = Socket::writeHttpResponseTo(page_stream, url);
70  if (!res) {
71  cerr << "Failed to download page at given url " << url.full() << endl;
72  return false;
73  }
74 
75  // Perform regex search
76  auto page = page_stream.str();
77  static std::regex url_regex(R"(.*src=["']([^"']*?(?:jpg|png|bmp|gif|pnm|JPG|PNG|BMP|GIF|PNM))["'].*)");
78  std::match_results<std::string::const_iterator> match;
79 
80  while (std::regex_search(page, match, url_regex)) {
81  auto image_url = Socket::Url::from_relative(url, match[1].str());
82 
83  std::string file_name(image_url.path_);
84  std::replace(file_name.begin(), file_name.end(), '/', '_');
85  utils::replace_all(file_name, "%2", "&");
86  std::cout << "Downloading " << file_name << std::endl;
87 
88  auto result_file = output;
89  result_file.append(file_name);
90  auto tmp_file = result_file;
91  tmp_file += ".tmp";
92  std::ofstream outfile(tmp_file, std::ios_base::trunc | std::ios_base::out | std::ios_base::binary);
93  res = Socket::writeHttpResponseTo(outfile, image_url);
94  if (res) {
95  rename(tmp_file, result_file);
96  } else {
97  remove(tmp_file);
98  std::cerr << "\tFailed to download: " << url.full() << endl;
99  }
100 
101  page = match.suffix();
102  }
103  return true;
