259  inline HttpParsedResponse receive(HttpHeaderParser::Callback write_back) {
260  if (socketId_ == 0) {
261  throw std::logic_error(from_parts("HttpSocket::", __func__,
262  ": accept called on a bad socket object"));
263  }
264 
265  // Closes the write half of the TCP stream socket (if not using SSL)
266  if (!cSSL_) {
267  if (::shutdown(getSocketId(), SHUT_WR) != 0) {
268  throw std::domain_error(
269  from_parts("HTTPProtocol::", __func__, ": shutdown: critical error: ", strerror(errno)));
270  }
271  }
272 
273  HttpHeaderParser parser;
274  std::size_t dataRead = 0;
275  while (dataRead < this->buffer_.size()) {
276  // Append to the internal buffer until EOF or the buffer is full
277  std::size_t get = read_from_socket(dataRead);
278  if (get == static_cast<std::size_t>(-1)) {
279  switch (errno) {
280  case 0:
281  break;
282  case EBADF:
283  case EFAULT:
284  case EINVAL:
285  case ENXIO: {
286  // Fatal error. Programming bug
287  throw std::domain_error(
288  from_parts("HttpSocket::", __func__, ": read: critical error: ",
289  strerror(errno)));
290  }
291  case EIO:
292  case ENOBUFS:
293  case ENOMEM: {
294  // Resource acquisition failure or device error
295  throw std::runtime_error(
296  from_parts("HttpSocket::", __func__, ": read: resource failure: ",
297  strerror(errno)));
298  }
299  case EINTR:
300  // TODO: Check for user interrupt flags.
301  // Beyond the scope of this project
302  // so continue normal operations.
303  case ETIMEDOUT:
304  case EAGAIN: {
305  // Temporary error.
306  // Simply retry the read.
307  continue;
308  }
309  case ECONNRESET:
310  case ENOTCONN: {
311  // Connection broken.
312  // Return the data we have available and exit
313  // as if the connection was closed correctly.
314  get = 0;
315  break;
316  }
317  default: {
318  throw std::runtime_error(
319  from_parts("HttpSocket::", __func__, ": read: returned -1:",
320  strerror(errno)));
321  }
322  }
323  }
324  if (get == 0) {
325  break;
326  }
327  dataRead += get;
328  if (parser.parse(this->buffer_.data(), dataRead, write_back)) {
329  dataRead = 0;
330  }
331  if (parser.receive_done()) break;
332  if (!parser.has_parsed() && this->buffer_.size() <= dataRead) {
333  throw std::runtime_error(
334  from_parts("HttpSocket::", __func__,
335  ": Did not receive HTTP status line and headers",
336  strerror(errno)));
337  }
338  }
339  return parser.parsed_header();
340  }
