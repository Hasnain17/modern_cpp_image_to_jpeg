98  // CPP: Compile time huffman table computation
99  constexpr std::array<BitCode, 256> generateHuffmanTable(const uint8_t numCodes[16], const uint8_t *values) {
100  std::array<BitCode, 256> result{};
101 
102  // process all bitsizes 1 thru 16, no JPEG Huffman code is allowed to exceed 16 bits
103  uint16_t huffmanCode = 0;
104  for (auto numBits = 1; numBits <= 16; numBits++) {
105  // ... and each code of these bitsizes
106  for (auto i = 0;
107  i < numCodes[numBits - 1]; i++) // note: numCodes array starts at zero, but smallest bitsize is 1
108  result[*values++] = BitCode(huffmanCode++, numBits);
109 
110  // next Huffman code needs to be one bit wider
111  huffmanCode <<= 1u;
112  }
113 
114  return result;
115  }

126  int16_t encode_block(BitWriter<> &writer, std::array<float, 8 * 8>& block, const std::array<float, 8 * 8>& scaled,
127  int16_t lastDC,
128  const std::array<BitCode, 256>& huffmanDC, const std::array<BitCode, 256>& huffmanAC,
129  const BitCode *codewords) noexcept {
130 
131  // "linearize" the 8x8 block, treat it as a flat array of 64 floats
132  auto block64 = block.data();
133 
134  // DCT: rows
135  for (auto offset = 0; offset < 8; offset++)
136  DCT<true>(block64 + offset * 8);
137  // DCT: columns
138  for (auto offset = 0; offset < 8; offset++)
139  DCT<false>(block64 + offset * 1);
140 
141  // scale
142  for (auto i = 0; i < 8 * 8; i++)
143  block64[i] *= scaled[i];
144 
145  // encode DC (the first coefficient is the "average color" of the 8x8 block)
146  auto DC = int(std::nearbyint(block64[0]));
147 
148  // quantize and zigzag the other 63 coefficients
149  auto posNonZero = 0; // find last coefficient which is not zero (because trailing zeros are encoded differently)
150  int16_t quantized[8 * 8];
151  for (auto i = 1; i < 8 * 8; i++) // start at 1 because block64[0]=DC was already processed
152  {
153  auto value = block64[ZigZagInv[i]];
154  quantized[i] = int(std::nearbyint(value));
155  // remember offset of last non-zero coefficient
156  if (quantized[i] != 0)
157  posNonZero = i;
158  }
159 
160  // same "average color" as previous block ?
161  auto diff = DC - lastDC;
162  if (diff == 0)
163  writer << huffmanDC[0x00]; // yes, write a special short symbol
164  else {
165  auto bits = codewords[diff]; // nope, encode the difference to previous block's average color
166  writer << huffmanDC[bits.numBits] << bits;
167  }
168 
169  // encode ACs (quantized[1..63])
170  auto offset = 0; // upper 4 bits count the number of consecutive zeros
171  // quantized[0] was already written, skip all trailing zeros, too
172  // zeros are encoded in a special way
173  for (auto i = 1; i <= posNonZero; i++) {
174  // found another zero ?
175  while (quantized[i] == 0) {
176  offset += 0x10; // add 1 to the upper 4 bits
177  // split into blocks of at most 16 consecutive zeros
178  if (offset > 0xF0) // remember, the counter is in the upper 4 bits, 0xF = 15
179  {
180  writer << huffmanAC[0xF0]; // 0xF0 is a special code for "16 zeros"
181  offset = 0;
182  }
183  i++;
184  }
185 
186  auto encoded = codewords[quantized[i]];
187  // combine number of zeros with the number of bits of the next non-zero value
188  writer << huffmanAC[offset + encoded.numBits] << encoded; // and the value itself
189  offset = 0;
190  }
191 
192  // send end-of-block code (0x00), only needed if there are trailing zeros
193  if (posNonZero < 8 * 8 - 1) // = 63
194  writer << huffmanAC[0x00];
195 
196  return DC;
197  }

218  0xFF, 0xE0, // JFIF APP0 tag

235  // each table has 64 entries and is preceded by an ID byte
236  // length: 65 bytes per table + 2 bytes for this length field
237  bitWriter.addMarker(0xDB_bn, 2 + (isRGB ? 2 : 1) * (1 + 8 * 8));
238

241  // second quantization table, only relevant for color images

245  // write image infos (SOF0 - start of frame)
246  // length: 6 bytes general info + 3 per channel + 2 bytes for this length field
247  bitWriter.addMarker(0xC0_bn, 2 + 6 + 3 * numComponents);

252  << (height >> 8u) << (height & 0xFFu)
253  << (width >> 8u) << (width & 0xFFu);

256  bitWriter << numComponents; // 1 component (grayscale, Y only) or 3 components (Y,Cb,Cr)
257  for (uint8_t id = 1; id <= numComponents; id++)
258  bitWriter << id // component ID (Y=1, Cb=2, Cr=3)
259  // bitmasks for sampling: highest 4 bits: horizontal, lowest 4 bits: vertical
260  << (id == 1 && downsample ? 0x22
261  : 0x11) // 0x11 is default YCbCr 4:4:4 and 0x22 stands for YCbCr 4:2:0
262  << (id == 1 ? 0 : 1); // use quantization table 0 for Y, table 1 for Cb and Cr

268  1 + 16 + 162; // for the AC luminance (chrominance)
269  // DHT marker - define Huffman tables; 2 bytes for the length field, store chrominance only if needed
270  bitWriter.addMarker(0xC4_bn, isRGB ? (2 + len + len) : (2 + len));
271

273  bitWriter << 0x00_bn // highest 4 bits: 0 => DC, lowest 4 bits: 0 => Y (baseline)
274  << DcLuminanceCodesPerBitsize

276  bitWriter << 0x10_bn // highest 4 bits: 1 => AC, lowest 4 bits: 0 => Y (baseline)
277  << AcLuminanceCodesPerBitsize
278  << AcLuminanceValues;

281  if (isRGB) {

283  bitWriter << 0x01_bn // highest 4 bits: 0 => DC, lowest 4 bits: 1 => Cr,Cb (baseline)
284  << DcChrominanceCodesPerBitsize
285  << DcChrominanceValues;

287  << AcChrominanceCodesPerBitsize

292  // start of scan (there is only a single scan for baseline JPEGs)
293  // 2 bytes for the length field, 1 byte for number of components,
294  // then 2 bytes for each component and 3 bytes for spectral selection
295  bitWriter.addMarker(0xDA_bn, 2 + 1 + 2 * numComponents + 3);
296 
297  // assign Huffman tables to each component
298  bitWriter << numComponents;
299  for (auto id = 1; id <= numComponents; id++)

301  // Y: tables 0 for DC and AC; Cb + Cr: tables 1 for DC and AC
302  bitWriter << id << (id == 1 ? 0x00_bn : 0x11_bn);

305  // spectral selection: must be from 0 to 63; successive approximation must be 0

311  const auto maxHeight = height - 1; // "bottom line"
312

327  for (auto mcuX = 0; mcuX < width; mcuX += mcuSize) {

378  // average/downsample chrominance of four pixels while respecting the image borders

395  auto down = pixelPos + rowStep;

412  column += 2;

419  }
